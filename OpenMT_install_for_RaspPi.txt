# Instructions for installing and configuring OpenMT on a Raspberry Pi
======================================================================
# NOTE:
The current software will only run on MMDVM or MMDVM_HS firmware.
If running M17 service you will need the firmware modified for M17.
Software tested on Raspberry Pi 2 model B and Pi 3 model B.

sudo apt update
sude apt upgrade

sudo raspi-config
# Goto Interface Options/Serial Port
# Login shell = No , Enable serial hardware = Yes
# Select Finish to exit.

# Edit /boot/firmware/cmdline.txt and add dwc_otg.lpm_enable=0, elevator=deadline and noswap.
# Edit /boot/firmware/config.txt and add dtoverlay=miniuart-bt to [all] section.

sudo apt install git
sudo apt install nginx
sudo apt install php-fpm
sudo apt install mariadb-server mariadb-client libmariadbd-dev
sudo apt install php
sudo apt install php-mysql
sudo apt install php-curl
sudo apt install build-essential
sudo apt install cmake

git clone https://github.com/kd0oss/OpenMT.git

# Set mariadb server to bind to all network interfaces.
sudo nano /etc/mysql/mariadb.conf.d/50-server.cnf
# Change bind-address to 0.0.0.0 from 127.0.0.1

# Configure databases.
sudo mysql -uroot
create database modem_host;
create user 'openmt'@'%' identified by 'dvpass123';
grant all privileges on modem_host.* to 'openmt'@'%';
flush privileges;
quit

# Upload schemas to databases.
cd OpenMT
# Import data into database tables.
# Password is "dvpass123" when asked.
mysql -u openmt -p modem_host < db/openmt.sql

# Install web based dashboard.
sudo rm -R /var/www/html
sudo mkdir /var/www/dashboard
sudo cp dashboard/www/* /var/www/dashboard

# Create database permissions file.
sudo nano /etc/openmt.ini
# Add these lines.
[database]
passwd=dvpass123
host=localhost
# Ctrl+x to save and close editor.

# Update Reflectors and DMR Id's.
cd OpenMT
php -q updateHosts.php

# Update /etc/nginx/sites-available/default to look like this:
sudo nano /etc/nginx/sites-available/default
# NOTE: In the line starting with fastcgi_pass change php version to match installed version.
server {
        listen 80 default_server;
        listen [::]:80 default_server;

        # SSL configuration
        #
        # listen 443 ssl default_server;
        # listen [::]:443 ssl default_server;
        #
        # Note: You should disable gzip for SSL traffic.
        # See: https://bugs.debian.org/773332
        #
        # Read up on ssl_ciphers to ensure a secure configuration.
        # See: https://bugs.debian.org/765782
        #
        # Self signed certs generated by the ssl-cert package
        # Don't use them in a production server!
        #
        # include snippets/snakeoil.conf;

        root /var/www/dashboard;

        # Add index.php to the list if you are using PHP
        index dashboard.html dashboard.htm;

        server_name _;

        location / {
                # First attempt to serve request as file, then
                # as directory, then fall back to displaying a 404.
                try_files $uri $uri/ =404;
        }

        # pass PHP scripts to FastCGI server
        #
        location ~ \.php$ {
                include snippets/fastcgi-php.conf;

                # With php-fpm (or other unix sockets):
                fastcgi_pass unix:/run/php/php8.4-fpm.sock;
                # With php-cgi (or other tcp sockets):
             #   fastcgi_pass 127.0.0.1:9000;
        }

        # deny access to .htaccess files, if Apache's document root
        # concurs with nginx's one
        #
        location ~ /\.ht {
                deny all;
        }
}

# Enable mysql by editing /etc/php/{PHP version}/fpm/php.ini
sudo nano /etc/php/{PHP version}/fpm/php.ini
# Search for extension=mysqli and uncomment it by removing the leading semicolon.
# Save changes.

# Now reboot.

# Login to RaspPI and cd to OpenMT.
cd OpenMT

######## NOTE: As you enter each executable directory compile using. #####
mkdir build
cd build
cmake ..
make
##########################################################################

# Then cd to OpenMT/modem and start modem host using
cd OpenMT/modem
sudo ./Modem_Host   # NOTE: sudo needed for access to /dev/ttyAMA0

# If setup properly the host should report no errors.
# You can optionally add -x at startup to display more debug info.
sudo ./Modem_Host -x

# If all is well you will want to start the host in daemon mode to free up the terminal.
ctrl+c to exit host
sudo ./Modem_Host -d
The initial run of the host will populate the database with default configuration.

# Point a browser to the RaspPI address and confirm Dashboard is accessable.
# Select Settings and a login box should appear. User name = admin, password = admin.
# Update info in General and Modem tabs to match your setup.
# Ignore modem config error, if encountered, for now.
# Restart Modem_Host to use update configuration.
# You will need to do this for each protocol service after running for fist time.

# Now to start M17 mode cd back to OpenMT directory then to protocols/m17 sub-directory.
cd ..
cd protocols/m17

# Start M17 host service. Same start up flags apply as modem host.
./m17_host

# If no errors stop and run as daemon.
ctrl+c to exit service
./m17_host -d

# Start gateway if desired. Save start up flags apply.
cd gateway
./m17_gateway -d

# M17 mode should now appear in the Dashboard.
# If reflectors have be loaded into database you can go to Settings/Mode for link/unlink.

# The only other protocol avaiable at the moment is DSTAR.
# NOTE: DSTAR gateway requires IRCDDBGateway package from G4KLX GitHub.
# Protocols and gateways can be run concurrently.
